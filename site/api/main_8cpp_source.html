67// ✅ Неблокирующая задача мониторинга кнопки сброса
68void resetButtonTask(void* /*parameter*/)
69{
70 pinMode(RESET_BUTTON_PIN, INPUT_PULLUP);
71 static unsigned long buttonPressTime = 0;
72 static bool buttonPressed = false;
73
74 while (true)
75 {
76 const bool currentState = (digitalRead(RESET_BUTTON_PIN) == LOW);
77
78 if (currentState && !buttonPressed)
79 {
80 // Кнопка только что нажата
81 buttonPressed = true;
82 buttonPressTime = millis();
83 logWarn("Кнопка сброса нажата! Сброс настроек через 2 сек...");
84 }
85 else if (!currentState && buttonPressed)
86 {
87 // Кнопка отпущена раньше времени
88 buttonPressed = false;
89 DEBUG_PRINTLN("Кнопка сброса отпущена");
90 }
91 else if (currentState && buttonPressed && (millis() - buttonPressTime >= BUTTON_HOLD_TIME_MS))
92 {
93 // Кнопка удерживалась 2 секунды
94 logError("Выполняется сброс настроек!");
95 resetConfig();
96 ESP.restart();
97 }
98
99 // ✅ Неблокирующая задержка - проверяем кнопку каждые 50мс
100 vTaskDelay(50 / portTICK_PERIOD_MS);
101 }
102}
103}

120 * Нужно успеть до истечения тайм-аутa загрузчика (~5 с).

122 const esp_partition_t* runningNow = esp_ota_get_running_partition();
123 esp_ota_img_states_t otaStateNow;
124 if (esp_ota_get_state_partition(runningNow, &otaStateNow) == ESP_OK && otaStateNow == ESP_OTA_IMG_PENDING_VERIFY)

126 logSystem("OTA image pending verify → подтверждаем (ранний этап)");

129 logSuccess("OTA image подтверждена, откат отменён");

227// ✅ Неблокирующий главный цикл с оптимизированными интервалами
228void loop()
229{
230 const unsigned long currentTime = millis();
231 esp_task_wdt_reset();
232
233 // Обновление NTP каждые 6 часов
234 if (timeClient != nullptr && millis() - lastNtpUpdate > 6 * 3600 * 1000)
235 {
236 timeClient->forceUpdate();
237 lastNtpUpdate = millis();
238 logSystemSafe("\1", timeClient->isTimeSet() ? "OK" : "не удалось");
239 }
240
241 // ✅ Вывод статуса системы каждые 30 секунд (неблокирующий)
242 if (currentTime - lastStatusPrint >= STATUS_PRINT_INTERVAL)
243 {
244 logPrintHeader("СТАТУС СИСТЕМЫ", LogColor::GREEN);
245
246 logUptime();
247 logMemoryUsage();
248 logWiFiStatus();
249 logSystemSafe("\1", config.flags.useRealSensor ? "РЕАЛЬНЫЙ" : "ЭМУЛЯЦИЯ");
250
251 // Статус данных датчика
252 if (sensorData.valid)
253 {
254 logDataSafe("\1", (currentTime - sensorData.last_update) / 1000.0);
255 }
256 else
257 {
258 logWarn("Данные датчика недоступны");
259 }
260
261 logPrintSeparator("─", 60);
262 lastStatusPrint = currentTime;
263 }
264
265 // === Проверяем наличие новых данных датчика (НАСТРАИВАЕМО v2.3.0) ===
266 if (sensorData.valid && (currentTime - lastDataPublish >= config.sensorReadInterval))
267 {
268 pendingMqttPublish = true;
269 pendingThingspeakPublish = true;
270 lastDataPublish = currentTime;
271 DEBUG_PRINTLN("[BATCH] Новые данные помечены для групповой отправки");
272 }
273
274 // ✅ Групповая отправка MQTT (настраиваемо v2.3.0)
275 if (pendingMqttPublish && (currentTime - mqttBatchTimer >= config.mqttPublishInterval))
276 {
277 publishSensorData();
278 pendingMqttPublish = false;
279 mqttBatchTimer = currentTime;
280 DEBUG_PRINTLN("[BATCH] MQTT данные отправлены группой");
281 }
282
283 // ✅ Групповая отправка ThingSpeak (настраиваемо v2.3.0)
284 if (pendingThingspeakPublish && (currentTime - thingspeakBatchTimer >= config.thingSpeakInterval))
285 {
286 const bool tsOk = sendDataToThingSpeak();
287 if (tsOk)
288 {
289 thingspeakBatchTimer = currentTime; // Сбрасываем таймер только при успешной отправке
290 DEBUG_PRINTLN("[BATCH] ThingSpeak данные отправлены группой");
291 }
292 else
293 {
294 DEBUG_PRINTLN("[BATCH] ThingSpeak отправка не удалась, повтор через следующий интервал");
295 }
296 pendingThingspeakPublish = false;
297 }
298
299 // ✅ Управление MQTT (каждые 100мс)
300 static unsigned long lastMqttCheck = 0;
301 if (currentTime - lastMqttCheck >= 100)
302 {
303 handleMQTT();
304 lastMqttCheck = currentTime;
305 }
306
307 // ✅ Управление WiFi (каждые 20 мс для более высокой отзывчивости веб-интерфейса)
308 static unsigned long lastWiFiCheck = 0;
309 if (currentTime - lastWiFiCheck >= 20)
310 {
311 handleWiFi();
312 lastWiFiCheck = currentTime;
313 }
314
315 // Проверяем OTA раз в час (или при принудительной проверке)
316 static unsigned long lastOtaCheck = 0;
317 if (config.flags.autoOtaEnabled && (currentTime - lastOtaCheck >= 3600000UL))
318 {
319 handleOTA();
320 lastOtaCheck = currentTime;
321 }
322
323 // ✅ Минимальная задержка для стабильности (10мс вместо 100мс)
324 vTaskDelay(10 / portTICK_PERIOD_MS);
325}
