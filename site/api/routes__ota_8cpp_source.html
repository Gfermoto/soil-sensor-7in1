343static void sendOtaStatusJson()
344{
345 StaticJsonDocument<OTA_JSON_DOC_SIZE> doc;
346
347 // Если идёт локальная загрузка, показываем её статус
348 if (isLocalUploadActive)
349 {
350 if (localUploadTotal > 0)
351 {
352 const size_t percent = (localUploadProgress * 100) / localUploadTotal;
353 doc["status"] = "local " + String(percent) + "%";
354 }
355 else
356 {
357 doc["status"] = "local " + String(localUploadProgress / 1024) + "KB";
358 }
359 doc["localUpload"] = true;
360 doc["progress"] = localUploadProgress;
361 doc["total"] = localUploadTotal;
362 }
363 else
364 {
365 doc["status"] = getOtaStatus();
366 doc["localUpload"] = false;
367 }
368
369 doc["version"] = JXCT_VERSION_STRING;
370 String json;
371 serializeJson(doc, json);
372 webServer.send(HTTP_OK, HTTP_CONTENT_TYPE_JSON, json);
373}

375// NOLINTNEXTLINE(misc-use-anonymous-namespace)
376static void handleFirmwareUpload()
377{
378 HTTPUpload& upload = webServer.upload();
379
380 if (upload.status == UPLOAD_FILE_START)
381 {
382 isLocalUploadActive = true;
383 localUploadProgress = 0;
384 localUploadTotal = upload.totalSize;
385 localUploadStatus = "uploading";
386
387 logSystemSafe("\1", upload.filename.c_str(), upload.totalSize);
388
389 if (!Update.begin(upload.totalSize == 0 ? UPDATE_SIZE_UNKNOWN : upload.totalSize))
390 {
391 logError("[OTA] Update.begin() failed");
392 Update.printError(Serial);
393 isLocalUploadActive = false;
394 localUploadStatus = "error";
395 // НЕ отправляем ответ здесь - это приведёт к обрыву загрузки
396 }
397 }
398 else if (upload.status == UPLOAD_FILE_WRITE)
399 {
400 if (Update.write(upload.buf, upload.currentSize) != upload.currentSize)
401 {
402 logErrorSafe("\1", upload.currentSize);
403 Update.printError(Serial);
404 isLocalUploadActive = false;
405 localUploadStatus = "error";
406 }
407 else
408 {
409 localUploadProgress += upload.currentSize;
410
411 // Логируем прогресс каждые 64KB
412 static size_t lastLogged = 0;
413 if (localUploadProgress - lastLogged > OTA_PROGRESS_LOG_THRESHOLD)
414 {
415 logSystemSafe("\1", localUploadProgress);
416 lastLogged = localUploadProgress;
417 }
418 }
419 }
420 else if (upload.status == UPLOAD_FILE_END)
421 {
422 logSystemSafe("\1", localUploadProgress);
423 localUploadStatus = "verifying";
424
425 if (Update.end(true)) // true = устанавливать как boot partition
426 {
427 if (Update.isFinished())
428 {
429 logSuccess("[OTA] Файл принят успешно, перезагрузка через 2 сек");
430 localUploadStatus = "success";
431 isLocalUploadActive = false;
432 webServer.send(HTTP_OK, HTTP_CONTENT_TYPE_JSON, R"({"ok":true})");
433 delay(OTA_DELAY_MS);
434 ESP.restart();
435 }
436 else
437 {
438 logError("[OTA] Update не завершён");
439 Update.printError(Serial);
440 isLocalUploadActive = false;
441 localUploadStatus = "error";
442 webServer.send(HTTP_OK, HTTP_CONTENT_TYPE_JSON, R"({"ok":false,"error":"not_finished"})");
443 }
444 }
445 else
446 {
447 logError("[OTA] Update.end() failed");
448 Update.printError(Serial);
449 isLocalUploadActive = false;
450 localUploadStatus = "error";
451 webServer.send(HTTP_OK, HTTP_CONTENT_TYPE_JSON, R"({"ok":false,"error":"end_failed"})");
452 }
453 }
454 else if (upload.status == UPLOAD_FILE_ABORTED)
455 {
456 logError("[OTA] Загрузка прервана");
457 Update.abort();
458 isLocalUploadActive = false;
459 localUploadStatus = "aborted";
460 webServer.send(HTTP_OK, HTTP_CONTENT_TYPE_JSON, R"({"ok":false,"error":"aborted"})");
461 }
462}
