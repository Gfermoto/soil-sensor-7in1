19void SensorCompensationService::applyCompensation(SensorData& data, SoilType soilType) {
20 logDebugSafe("SensorCompensationService: Применение компенсации для типа почвы %d", static_cast<int>(soilType));
21
22 // Применяем компенсацию к каждому параметру

36float SensorCompensationService::correctEC(float ec25, float temperature, float humidity, SoilType soilType) {
37 if (!validateCompensationInputs(temperature, humidity, soilType)) {
38 logDebugSafe("SensorCompensationService: Недопустимые входные данные для компенсации EC");
39 return ec25;
40 }
41
42 // Получаем коэффициенты Арчи для типа почвы
43 ArchieCoefficients coeffs = getArchieCoefficients(soilType);
44
45 // Температурная компенсация по модели Арчи
46 float tempFactor = calculateECTemperatureFactor(temperature);
47
48 // Влажностная компенсация по модели Арчи
49 float humidityFactor = calculateECHumidityFactor(humidity, soilType);
50
51 // Применяем модель Арчи: EC = EC0 * (θ/θ0)^m * (T/T0)^n
52 float compensatedEC = ec25 * pow(humidityFactor, coeffs.m) * pow(tempFactor, coeffs.n);
53
54 logDebugSafe("SensorCompensationService: EC скорректирован %.2f → %.2f", ec25, compensatedEC);
55 return compensatedEC;
56}

58float SensorCompensationService::correctPH(float phRaw, float temperature) {
59 if (temperature < -50.0F || temperature > 100.0F) {
60 logDebugSafe("SensorCompensationService: Недопустимая температура для компенсации pH: %.2f", temperature);
61 return phRaw;
62 }
63
64 // Уравнение Нернста для температурной компенсации pH
65 float tempKelvin = temperatureToKelvin(temperature);
66 float tempCorrection = (R * tempKelvin / F) * log(tempKelvin / T0) / log(10.0F);
67
68 float compensatedPH = phRaw + tempCorrection;
69
70 logDebugSafe("SensorCompensationService: pH скорректирован %.2f → %.2f", phRaw, compensatedPH);
71 return compensatedPH;
72}

74void SensorCompensationService::correctNPK(float temperature, float humidity, NPKReferences& npk, SoilType soilType) {
75 if (!validateCompensationInputs(temperature, humidity, soilType)) {
76 logDebugSafe("SensorCompensationService: Недопустимые входные данные для компенсации NPK");
77 return;
78 }
79
80 // Получаем параметры почвы
81 SoilParameters params = getSoilParameters(soilType);
82
83 // Алгоритм FAO 56 для компенсации NPK
84 float tempFactor = 1.0F + 0.02F * (temperature - 25.0F);
85 float humidityFactor = 1.0F + 0.01F * (humidity - params.fieldCapacity * 100.0F);
86
87 // Применяем компенсацию с учетом пористости почвы
88 float porosityFactor = params.porosity / 0.45F; // Нормализация к стандартной пористости
89
90 npk.nitrogen *= tempFactor * humidityFactor * porosityFactor;
91 npk.phosphorus *= tempFactor * humidityFactor * porosityFactor;
92 npk.potassium *= tempFactor * humidityFactor * porosityFactor;
93
94 logDebugSafe("SensorCompensationService: NPK скорректирован N:%.2f P:%.2f K:%.2f",
95 npk.nitrogen, npk.phosphorus, npk.potassium);
96}

114bool SensorCompensationService::validateCompensationInputs(float temperature, float humidity, SoilType soilType) const {
115 if (temperature < -50.0F || temperature > 100.0F) {
116 return false;
117 }
118
119 if (humidity < 0.0F || humidity > 100.0F) {
120 return false;
121 }
122
123 // Проверяем, что тип почвы валиден
124 switch (soilType) {
125 case SoilType::SAND:
126 case SoilType::LOAM:
127 case SoilType::PEAT:
128 case SoilType::CLAY:
129 case SoilType::SANDPEAT:
130 return true;
131 default:
132 return false;
133 }
134}

153 // Коэффициенты Арчи для разных типов почвы
154 // Источник: Archie, G.E. (1942). The electrical resistivity log as an aid in determining some reservoir characteristics.
155
156 archieCoefficients[SoilType::SAND] = ArchieCoefficients(1.3F, 2.0F, 0.35F);

159 archieCoefficients[SoilType::CLAY] = ArchieCoefficients(2.0F, 2.5F, 0.50F);

162 logDebugSafe("SensorCompensationService: Коэффициенты Арчи инициализированы");

188float SensorCompensationService::calculateECHumidityFactor(float humidity, SoilType soilType) const {
189 // Влажностная компенсация для EC
190 // Нормализация к полевой влагоемкости
191 SoilParameters params = getSoilParameters(soilType);
192 float fieldCapacityPercent = params.fieldCapacity * 100.0F;
193
194 return 1.0F + 0.01F * (humidity - fieldCapacityPercent);
195}
