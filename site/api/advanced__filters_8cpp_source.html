24 float smoothed_value;
25 bool initialized;

27 ExponentialSmoothingState() : smoothed_value(0.0F), initialized(false) {}
28};

38float applyExponentialSmoothing(float new_value, ExponentialSmoothingState& state, float alpha) {
39 if (!state.initialized) {
40 state.smoothed_value = new_value;
41 state.initialized = true;
42 return new_value;
43 }
44
45 // Экспоненциальное сглаживание: S_t = α * X_t + (1-α) * S_{t-1}
46 state.smoothed_value = alpha * new_value + (1.0F - alpha) * state.smoothed_value;
47 return state.smoothed_value;
48}

80void updateStatistics(float new_value, StatisticsBuffer& buffer) {
81 // Добавляем новое значение
82 buffer.values[buffer.index] = new_value;
83 buffer.index = (buffer.index + 1) % STATISTICS_WINDOW_SIZE;
84
85 if (buffer.filled < STATISTICS_WINDOW_SIZE) {
86 buffer.filled++;
87 }
88
89 // Вычисляем среднее
90 float sum = 0.0F;
91 for (uint8_t i = 0; i < buffer.filled; ++i) {
92 sum += buffer.values[i];
93 }
94 buffer.mean = sum / static_cast<float>(buffer.filled);
95
96 // Вычисляем стандартное отклонение
97 float variance_sum = 0.0F;
98 for (uint8_t i = 0; i < buffer.filled; ++i) {
99 float diff = buffer.values[i] - buffer.mean;
100 variance_sum += diff * diff;
101 }
102 buffer.std_dev = sqrt(variance_sum / static_cast<float>(buffer.filled));
103
104 // Минимальное стандартное отклонение для стабильности
105 if (buffer.std_dev < MIN_STANDARD_DEVIATION) {
106 buffer.std_dev = MIN_STANDARD_DEVIATION;
107 }
108
109 buffer.valid = (buffer.filled >= 5); // Минимум 5 значений для статистики
110}
111
112bool isOutlier(float value, const StatisticsBuffer& buffer, float threshold_multiplier) {
113 if (!buffer.valid) {
114 return false; // Недостаточно данных для определения выброса
115 }
116
117 float deviation = abs(value - buffer.mean);
118 float threshold = threshold_multiplier * buffer.std_dev;
119
120 return deviation > threshold;
121}

127struct KalmanFilter {
128 float x; // Состояние (оценка)
129 float P; // Ковариация ошибки оценки
130 float Q; // Шум процесса
131 float R; // Шум измерений
132 bool initialized;
133
134 KalmanFilter(float process_noise, float measurement_noise)
135 : x(0.0F), P(KALMAN_INITIAL_UNCERTAINTY), Q(process_noise), R(measurement_noise), initialized(false) {}
136
137 float update(float measurement) {
138 if (!initialized) {
139 x = measurement;
140 initialized = true;
141 return measurement;
142 }
143
144 // Предсказание
145 float P_pred = P + Q;
146
147 // Обновление
148 float K = P_pred / (P_pred + R); // Коэффициент Калмана
149 x = x + K * (measurement - x);
150 P = (1.0F - K) * P_pred;
151
152 return x;
153 }
154
155 void reset() {
156 x = 0.0F;
157 P = KALMAN_INITIAL_UNCERTAINTY;
158 initialized = false;
159 }
160};

174float applyCombinedFilter(float raw_value, FilterType type, bool enable_kalman, bool enable_adaptive) {
175 float filtered_value = raw_value;
176
177 // Специализированная фильтрация EC
178 if (type == FilterType::EC) {
179 filtered_value = applyECSpecializedFilter(raw_value);
180 }
181
182 // 1. Обновляем статистику для адаптивных порогов
183 if (enable_adaptive) {
184 StatisticsBuffer* buffer = nullptr;
185 switch (type) {
186 case FilterType::TEMPERATURE: buffer = &stats_temp; break;
187 case FilterType::HUMIDITY: buffer = &stats_hum; break;
188 case FilterType::EC: buffer = &stats_ec; break;
189 case FilterType::PH: buffer = &stats_ph; break;
190 case FilterType::NITROGEN: buffer = &stats_n; break;
191 case FilterType::PHOSPHORUS: buffer = &stats_p; break;
192 case FilterType::POTASSIUM: buffer = &stats_k; break;
193 }
194
195 if (buffer) {
196 updateStatistics(filtered_value, *buffer);
197
198 // Проверяем на выбросы
199 float threshold = config.outlierThreshold;
200
201 // Специальная обработка для EC - более строгие пороги
202 if (type == FilterType::EC) {
203 threshold = config.outlierThreshold * 0.7F; // Более строгий порог для EC
204
205 // Дополнительная проверка для EC - если значение слишком сильно отличается от предыдущего
206 if (buffer->filled >= 5) { // Нужно минимум 5 измерений
207 float last_value = buffer->values[(buffer->index - 1 + STATISTICS_WINDOW_SIZE) % STATISTICS_WINDOW_SIZE];
208 float change_percent = abs(filtered_value - last_value) / last_value * 100.0F;
209
210 // Если изменение больше 20% - считаем выбросом
211 if (change_percent > 20.0F) {
212 return buffer->mean;
213 }
214 }
215 }
216
217 if (isOutlier(filtered_value, *buffer, threshold)) {
218 // Возвращаем предыдущее значение вместо выброса
219 return buffer->mean;
220 }
221 }
222 }
223
224 // 2. Применяем экспоненциальное сглаживание
225 ExponentialSmoothingState* exp_state = nullptr;
226 float alpha = config.exponentialAlpha; // Базовый коэффициент
227
228 // Дифференцированные настройки для шумных параметров
229 switch (type) {
230 case FilterType::EC:
231 alpha = config.exponentialAlpha * 0.5F; // Очень агрессивное сглаживание для EC
232 break;
233 case FilterType::NITROGEN:
234 case FilterType::PHOSPHORUS:
235 case FilterType::POTASSIUM:
236 alpha = config.exponentialAlpha * 0.8F; // Умеренное сглаживание для NPK
237 break;
238 default:
239 alpha = config.exponentialAlpha; // Обычное сглаживание для остальных
240 break;
241 }
242
243 switch (type) {
244 case FilterType::TEMPERATURE: exp_state = &exp_smooth_temp; break;
245 case FilterType::HUMIDITY: exp_state = &exp_smooth_hum; break;
246 case FilterType::EC: exp_state = &exp_smooth_ec; break;
247 case FilterType::PH: exp_state = &exp_smooth_ph; break;
248 case FilterType::NITROGEN: exp_state = &exp_smooth_n; break;
249 case FilterType::PHOSPHORUS: exp_state = &exp_smooth_p; break;
250 case FilterType::POTASSIUM: exp_state = &exp_smooth_k; break;
251 }
252
253 if (exp_state) {
254 filtered_value = applyExponentialSmoothing(filtered_value, *exp_state, alpha);
255 }
256
257 // 3. Применяем фильтр Калмана (если включен)
258 if (enable_kalman) {
259 KalmanFilter* kalman = nullptr;
260 switch (type) {
261 case FilterType::TEMPERATURE: kalman = &kalman_temp; break;
262 case FilterType::HUMIDITY: kalman = &kalman_hum; break;
263 case FilterType::EC: kalman = &kalman_ec; break;
264 case FilterType::PH: kalman = &kalman_ph; break;
265 case FilterType::NITROGEN: kalman = &kalman_n; break;
266 case FilterType::PHOSPHORUS: kalman = &kalman_p; break;
267 case FilterType::POTASSIUM: kalman = &kalman_k; break;
268 }
269
270 if (kalman) {
271 filtered_value = kalman->update(filtered_value);
272 }
273 }
274
275 return filtered_value;
276}

282struct ECFilterState {
283 std::array<float, 10> recent_values; // Последние 10 значений
284 uint8_t index;
285 uint8_t filled;
286 float baseline; // Базовое значение
287 uint32_t last_spike_time; // Время последнего выброса
288 uint8_t spike_count; // Счетчик выбросов
289 bool baseline_valid;
290
291 ECFilterState() : index(0), filled(0), baseline(0.0F),
292 last_spike_time(0), spike_count(0), baseline_valid(false) {
293 recent_values.fill(0.0F);
294 }
295};
296
297static ECFilterState ec_filter_state;
298
299// Анализ паттерна выбросов EC
300bool isECSpikePattern(float new_value, float baseline) {
301 if (!ec_filter_state.baseline_valid) {
302 return false;
303 }
304
305 float spike_threshold = baseline * 0.15F; // 15% от базового значения
306 float spike_height = new_value - baseline;
307
308 // Проверяем, что это выброс вверх
309 if (spike_height < spike_threshold) {
310 return false;
311 }
312
313 // Проверяем периодичность (если выбросы происходят регулярно)
314 uint32_t current_time = millis();
315 uint32_t time_since_last = current_time - ec_filter_state.last_spike_time;
316
317 // Если выбросы происходят с интервалом 2-10 секунд - это паттерн
318 if (time_since_last > 2000 && time_since_last < 10000) {
319 ec_filter_state.spike_count++;
320 ec_filter_state.last_spike_time = current_time;
321
322 // Если за последние 30 секунд было больше 3 выбросов - это системная проблема
323 if (ec_filter_state.spike_count > 3) {
324 return true;
325 }
326 }
327
328 return false;
329}

331// Обновление базового значения EC
332void updateECBaseline(float new_value) {
333 if (!ec_filter_state.baseline_valid) {
334 ec_filter_state.baseline = new_value;
335 ec_filter_state.baseline_valid = true;
336 return;
337 }
338
339 // Медленное обновление базового значения (α = 0.1)
340 ec_filter_state.baseline = ec_filter_state.baseline * 0.9F + new_value * 0.1F;
341}

343// Специализированная фильтрация EC
344float applyECSpecializedFilter(float raw_value) {
345 // Обновляем историю значений
346 ec_filter_state.recent_values[ec_filter_state.index] = raw_value;
347 ec_filter_state.index = (ec_filter_state.index + 1) % 10;
348 if (ec_filter_state.filled < 10) ec_filter_state.filled++;
349
350 // Обновляем базовое значение
351 updateECBaseline(raw_value);
352
353 // Проверяем паттерн выбросов
354 if (isECSpikePattern(raw_value, ec_filter_state.baseline)) {
355 logSystemSafe("[EC_FILTER] Обнаружен паттерн выбросов: %.1f -> %.1f (база: %.1f)",
356 ec_filter_state.baseline, raw_value, ec_filter_state.baseline);
357 return ec_filter_state.baseline; // Возвращаем базовое значение
358 }
359
360 // Дополнительная проверка на аномальные скачки
361 if (ec_filter_state.filled >= 3) {
362 float prev_value = ec_filter_state.recent_values[(ec_filter_state.index - 2 + 10) % 10];
363 float change_percent = abs(raw_value - prev_value) / prev_value * 100.0F;
364
365 // Если изменение больше 25% - считаем выбросом
366 if (change_percent > 25.0F) {
367 logSystemSafe("[EC_FILTER] Аномальный скачок: %.1f -> %.1f (%.1f%%)",
368 prev_value, raw_value, change_percent);
369 return prev_value;
370 }
371 }
372
373 return raw_value;
374}

380void applyAdvancedFiltering(SensorData& data) {
381 if (!config.adaptiveFiltering && !config.kalmanEnabled) {
382 return; // Фильтрация отключена
383 }
384
385 // Применяем комбинированный фильтр ко всем параметрам
386 data.temperature = applyCombinedFilter(data.temperature, FilterType::TEMPERATURE,
387 config.kalmanEnabled, config.adaptiveFiltering);
388 data.humidity = applyCombinedFilter(data.humidity, FilterType::HUMIDITY,
389 config.kalmanEnabled, config.adaptiveFiltering);
390 data.ec = applyCombinedFilter(data.ec, FilterType::EC,
391 config.kalmanEnabled, config.adaptiveFiltering);
392 data.ph = applyCombinedFilter(data.ph, FilterType::PH,
393 config.kalmanEnabled, config.adaptiveFiltering);
394 data.nitrogen = applyCombinedFilter(data.nitrogen, FilterType::NITROGEN,
395 config.kalmanEnabled, config.adaptiveFiltering);
396 data.phosphorus = applyCombinedFilter(data.phosphorus, FilterType::PHOSPHORUS,
397 config.kalmanEnabled, config.adaptiveFiltering);
398 data.potassium = applyCombinedFilter(data.potassium, FilterType::POTASSIUM,
399 config.kalmanEnabled, config.adaptiveFiltering);
400}
