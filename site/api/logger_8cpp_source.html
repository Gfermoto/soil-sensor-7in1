18String getUptimeString()
19{
20 const unsigned long milliseconds = millis();
21 unsigned long seconds = milliseconds / 1000;
22 unsigned long minutes = seconds / 60;
23 unsigned long hours = minutes / 60;
24 const unsigned long days = hours / 24;
25
26 seconds %= 60;
27 minutes %= 60;
28 hours %= 24;
29
30 return String(days) + "д " + String(hours) + "ч " + String(minutes) + "м " + String(seconds) + "с";
31}
