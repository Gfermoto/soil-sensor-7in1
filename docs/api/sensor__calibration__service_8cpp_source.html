16void SensorCalibrationService::applyCalibration(SensorData& data, SoilProfile profile) {
17 logDebugSafe("SensorCalibrationService: Применение калибровки для профиля %d", static_cast<int>(profile));
18
19 // Сохраняем исходные значения
20 data.raw_temperature = data.temperature;

22 data.raw_ec = data.ec;
23 data.raw_ph = data.ph;
24 data.raw_nitrogen = data.nitrogen;
25 data.raw_phosphorus = data.phosphorus;

32 if (!table.temperaturePoints.empty()) {
33 data.temperature = applyCalibrationWithInterpolation(data.temperature, table.temperaturePoints);

35
36 if (!table.humidityPoints.empty()) {
37 data.humidity = applyCalibrationWithInterpolation(data.humidity, table.humidityPoints);

39
40 if (!table.ecPoints.empty()) {
41 data.ec = applyCalibrationWithInterpolation(data.ec, table.ecPoints);
42 }
43
44 if (!table.phPoints.empty()) {
45 data.ph = applyCalibrationWithInterpolation(data.ph, table.phPoints);
46 }
47
48 if (!table.nitrogenPoints.empty()) {

53 data.phosphorus = applyCalibrationWithInterpolation(data.phosphorus, table.phosphorusPoints);
54 }

86bool SensorCalibrationService::validateCalibrationData(const SensorData& data) {
87 // Проверяем диапазоны значений
88 if (data.temperature < -50.0F || data.temperature > 100.0F) {
89 logDebugSafe("SensorCalibrationService: Недопустимая температура: %.2f", data.temperature);
90 return false;
91 }
92
93 if (data.humidity < 0.0F || data.humidity > 100.0F) {
94 logDebugSafe("SensorCalibrationService: Недопустимая влажность: %.2f", data.humidity);
95 return false;
96 }
97
98 if (data.ec < 0.0F || data.ec > 10000.0F) {
99 logDebugSafe("SensorCalibrationService: Недопустимая EC: %.2f", data.ec);
100 return false;
101 }
102
103 if (data.ph < 0.0F || data.ph > 14.0F) {
104 logDebugSafe("SensorCalibrationService: Недопустимый pH: %.2f", data.ph);
105 return false;
106 }
107
108 if (data.nitrogen < 0.0F || data.nitrogen > 1000.0F) {
109 logDebugSafe("SensorCalibrationService: Недопустимый азот: %.2f", data.nitrogen);
110 return false;
111 }
112
113 if (data.phosphorus < 0.0F || data.phosphorus > 1000.0F) {
114 logDebugSafe("SensorCalibrationService: Недопустимый фосфор: %.2f", data.phosphorus);
115 return false;
116 }
117
118 if (data.potassium < 0.0F || data.potassium > 1000.0F) {
119 logDebugSafe("SensorCalibrationService: Недопустимый калий: %.2f", data.potassium);
120 return false;
121 }
122
123 return true;
124}

206float SensorCalibrationService::applyCalibrationWithInterpolation(float rawValue,
207 const std::vector<CalibrationPoint>& points) const {
208 if (points.empty()) {
209 return rawValue;
210 }
211
212 if (points.size() == 1) {
213 return points[0].referenceValue;
214 }
215
216 // Находим ближайшие точки для интерполяции
217 for (size_t i = 0; i < points.size() - 1; ++i) {
218 if (rawValue >= points[i].rawValue && rawValue <= points[i + 1].rawValue) {
219 return linearInterpolation(rawValue,
220 points[i].rawValue, points[i].referenceValue,
221 points[i + 1].rawValue, points[i + 1].referenceValue);
222 }
223 }
224
225 // Если значение вне диапазона, используем экстраполяцию
226 if (rawValue < points[0].rawValue) {
227 return points[0].referenceValue;
228 } else {
229 return points[points.size() - 1].referenceValue;
230 }
231}

233float SensorCalibrationService::linearInterpolation(float x, float x1, float y1, float x2, float y2) const {
234 if (x2 == x1) {
235 return y1;
236 }
237 return y1 + (y2 - y1) * (x - x1) / (x2 - x1);
238}

240bool SensorCalibrationService::parseCalibrationCSV(const String& csvData, CalibrationTable& table) {
241 // Простая реализация парсинга CSV
242 // В реальной реализации здесь был бы более сложный парсер
243
244 // Разбиваем на строки
245 int startPos = 0;
246 int endPos = csvData.indexOf('\n');
247
248 // Пропускаем заголовок
249 if (endPos != -1) {
250 startPos = endPos + 1;
251 endPos = csvData.indexOf('\n', startPos);
252 }
253
254 while (endPos != -1) {
255 String line = csvData.substring(startPos, endPos);
256 if (line.length() > 0) {
257 // Простой парсинг: sensor_type,raw_value,reference_value
258 int comma1 = line.indexOf(',');
259 int comma2 = line.indexOf(',', comma1 + 1);
260
261 if (comma1 != -1 && comma2 != -1) {
262 String sensorType = line.substring(0, comma1);
263 float rawValue = line.substring(comma1 + 1, comma2).toFloat();
264 float referenceValue = line.substring(comma2 + 1).toFloat();
265
266 CalibrationPoint point(rawValue, referenceValue);
267
268 if (sensorType == "temperature") {
269 table.temperaturePoints.push_back(point);
270 } else if (sensorType == "humidity") {
271 table.humidityPoints.push_back(point);
272 } else if (sensorType == "ec") {
273 table.ecPoints.push_back(point);
274 } else if (sensorType == "ph") {
275 table.phPoints.push_back(point);
276 } else if (sensorType == "nitrogen") {
277 table.nitrogenPoints.push_back(point);
278 } else if (sensorType == "phosphorus") {
279 table.phosphorusPoints.push_back(point);
280 } else if (sensorType == "potassium") {
281 table.potassiumPoints.push_back(point);
282 }
283 }
284 }
285
286 startPos = endPos + 1;
287 endPos = csvData.indexOf('\n', startPos);
288 }
289
290 // Проверяем валидность таблицы
291 table.isValid = !table.temperaturePoints.empty() || !table.humidityPoints.empty() ||
292 !table.ecPoints.empty() || !table.phPoints.empty() ||
293 !table.nitrogenPoints.empty() || !table.phosphorusPoints.empty() ||
294 !table.potassiumPoints.empty();
295
296 return table.isValid;
297}

299bool SensorCalibrationService::validateCalibrationPoints(const std::vector<CalibrationPoint>& points) const {
300 if (points.empty()) {
301 return true;
302 }
303
304 // Проверяем, что точки отсортированы по возрастанию
305 for (size_t i = 1; i < points.size(); ++i) {

307 return false;
308 }
309 }
310
311 return true;
