16{
17 unsigned long totalSeconds = millis() / 1000;
18 unsigned long hours = totalSeconds / 3600;
19 unsigned long minutes = (totalSeconds % 3600) / 60;
20 unsigned long seconds = totalSeconds % 60;
21
22 return String(hours) + ":" + (minutes < 10 ? "0" : "") + String(minutes) + ":" + (seconds < 10 ? "0" : "") +
23 String(seconds);
24}
