344bool checkResetButton()
345{
346 static unsigned long pressStart = 0;
347 static bool wasPressed = false;
348 const bool isPressed = digitalRead(static_cast<uint8_t>(WifiConstants::RESET_BUTTON_PIN)) == LOW;
349 if (isPressed && !wasPressed)
350 {
351 pressStart = millis();
352 wasPressed = true;
353 setLedFastBlink();
354 return false;
355 }
356
357 if (!isPressed && wasPressed)
358 {
359 wasPressed = false;
360 setLedBlink(static_cast<unsigned long>(WifiConstants::LED_SLOW_BLINK_INTERVAL));
361 return false;
362 }
363
364 if (isPressed && wasPressed)
365 {
366 if (millis() - pressStart >= static_cast<unsigned long>(WifiConstants::RESET_BUTTON_HOLD_TIME))
367 {
368 return true;
369 }
370 }
371 return false;
372}
