79// Современная C++ альтернатива с template parameter pack
80template <typename... Args>
81String formatLogMessageSafe(const char* format, Args&&... args)
82{
83 std::array<char, 512> buffer;
84 int result = snprintf(buffer.data(), buffer.size(), format, std::forward<Args>(args)...);
85 if (result < 0)
86 {
87 return String("LOG FORMAT ERROR");
88 }
89 if (result < static_cast<int>(buffer.size()))
90 {
91 return String(buffer.data());
92 }
93 buffer[buffer.size() - 1] = '\0';
94 return String(buffer.data());
95}
