212String formatUptime(unsigned long milliseconds)
213{
214 unsigned long seconds = milliseconds / 1000;
215 unsigned long minutes = seconds / 60;
216 unsigned long hours = minutes / 60;
217 unsigned long days = hours / 24;
218
219 seconds %= 60;
220 minutes %= 60;
221 hours %= 24;
222
223 String uptime = "";
224 if (days > 0) uptime += String(days) + "–¥ ";
225 if (hours > 0) uptime += String(hours) + "—á ";
226 if (minutes > 0) uptime += String(minutes) + "–º ";
227 uptime += String(seconds) + "—Å";
228
229 return uptime;
230}
