46bool validateConfigInput(bool checkRequired)
47{
48 // Валидация базовых полей
49 if (checkRequired)
50 {
51 // SSID всегда обязательно
52 if (!webServer.hasArg("ssid") || webServer.arg("ssid").length() == 0)
53 {
54 logWarn("Валидация: отсутствует SSID");
55 return false;
56 }
57
58 // В STA режиме проверяем дополнительные поля
59 if (currentWiFiMode == WiFiMode::STA)
60 {
61 // Если MQTT включен, проверяем обязательные поля
62 if (webServer.hasArg("mqtt_enabled"))
63 {
64 if (!webServer.hasArg("mqtt_server") || webServer.arg("mqtt_server").length() == 0)
65 {
66 logWarn("Валидация: MQTT включен, но отсутствует сервер");
67 return false;
68 }
69 }
70
71 // Если ThingSpeak включен, проверяем API ключ
72 if (webServer.hasArg("ts_enabled"))
73 {
74 if (!webServer.hasArg("ts_api_key") || webServer.arg("ts_api_key").length() == 0)
75 {
76 logWarn("Валидация: ThingSpeak включен, но отсутствует API ключ");
77 return false;
78 }
79 }
80 }
81 }
82
83 // Валидация форматов данных
84 if (!validateInterval("mqtt_port", 1, 65535, "MQTT порт")) return false;
85 if (!validateInterval("ntp_interval", 10000, 86400000, "NTP интервал")) return false;
86 if (!validateInterval("sensor_read", 1000, 300000, "интервал чтения датчика")) return false;
87 if (!validateInterval("mqtt_publish", 1000, 3600000, "интервал MQTT публикации")) return false;
88 if (!validateInterval("thingspeak_interval", 15000, 7200000, "интервал ThingSpeak")) return false;
89
90 logDebug("Валидация конфигурации прошла успешно");
91 return true;
92}

123void logWebRequest(const String& method, const String& uri, const String& clientIP)
124{
125 // Логирование только важных запросов, исключаем служебные
126 if (uri.startsWith("/sensor_json") || uri.startsWith(API_SENSOR))
127 {
128 // API запросы логируем на уровне DEBUG
129 logDebug("%s %s from %s", method.c_str(), uri.c_str(), clientIP.c_str());
130 }
131 else
132 {
133 // Обычные запросы на уровне INFO
134 logInfo("Web: %s %s from %s", method.c_str(), uri.c_str(), clientIP.c_str());
135 }
136}
