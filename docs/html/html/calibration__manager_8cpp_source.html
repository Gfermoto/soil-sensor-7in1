58bool loadTable(SoilProfile profile, CalibrationEntry* outBuffer, size_t maxEntries, size_t& outCount)
59{
60 outCount = 0;
61 if (!init()) return false;
62 const char* path = profileToFilename(profile);
63 File f = LittleFS.open(path, "r");
64 if (!f)
65 {
66 logWarn("Нет калибровочной таблицы %s", path);
67 return false;
68 }
69
70 String line;
71 while (f.available() && outCount < maxEntries)
72 {
73 line = f.readStringUntil('\n');
74 line.trim();
75 if (line.length() == 0) continue;
76 if (line[0] == '#') continue; // комментарий
77
78 // Пропускаем строку-заголовок (если обнаружены буквы)
79 if (!isDigit(line[0]) && line[0] != '-') continue;
80
81 int comma = line.indexOf(',');
82 if (comma < 0) continue;
83
84 float raw = line.substring(0, comma).toFloat();
85 float corr = line.substring(comma + 1).toFloat();
86 outBuffer[outCount++] = {raw, corr};
87 }
88
89 f.close();
90 logInfo("Загружено %d записей из %s", outCount, path);
91 return outCount > 0;
92}

111float applyCalibration(float rawValue, SoilProfile profile)
112{
113 // Если калибровочная таблица не загружена, возвращаем исходное значение
114 if (!hasTable(profile))
115 {
116 return rawValue;
117 }
118
119 // Загружаем калибровочную таблицу
120 constexpr size_t MAX_ENTRIES = 100;
121 CalibrationEntry entries[MAX_ENTRIES];
122 size_t entryCount;
123
124 if (!loadTable(profile, entries, MAX_ENTRIES, entryCount) || entryCount == 0)
125 {
126 return rawValue;
127 }
128
129 // Ищем ближайшие значения для интерполяции
130 float lowerRaw = entries[0].raw;
131 float lowerCorr = entries[0].corrected;
132 float upperRaw = entries[entryCount - 1].raw;
133 float upperCorr = entries[entryCount - 1].corrected;
134
135 // Ищем точное совпадение или ближайшие значения
136 for (size_t i = 0; i < entryCount; i++)
137 {
138 if (entries[i].raw == rawValue)
139 {
140 // Точное совпадение - применяем коэффициент
141 return rawValue * entries[i].corrected;
142 }
143
144 if (entries[i].raw < rawValue)
145 {
146 lowerRaw = entries[i].raw;
147 lowerCorr = entries[i].corrected;
148 }
149 else if (entries[i].raw > rawValue)
150 {
151 upperRaw = entries[i].raw;
152 upperCorr = entries[i].corrected;
153 break;
154 }
155 }
156
157 // Линейная интерполяция между ближайшими значениями
158 if (upperRaw > lowerRaw)
159 {
160 float ratio = (rawValue - lowerRaw) / (upperRaw - lowerRaw);
161 float interpolatedCoeff = lowerCorr + ratio * (upperCorr - lowerCorr);
162 return rawValue * interpolatedCoeff;
163 }
164 else
165 {
166 // Если нет интервала, используем ближайший коэффициент
167 return rawValue * lowerCorr;
168 }
169}
