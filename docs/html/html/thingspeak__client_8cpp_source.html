21static void trim(char* s)
22{
23 if (!s) return;
24 // Trim leading
25 char* p = s;
26 while (*p && isspace((unsigned char)*p)) ++p;
27 if (p != s) memmove(s, p, strlen(p) + 1);
28
29 // Trim trailing
30 size_t len = strlen(s);
31 while (len > 0 && isspace((unsigned char)s[len - 1])) s[--len] = '\0';
32}
