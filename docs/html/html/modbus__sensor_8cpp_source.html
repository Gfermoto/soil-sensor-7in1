108// Функция для расчета CRC16 Modbus
109uint16_t calculateCRC16(uint8_t* data, size_t length)
110{
111 uint16_t crc = 0xFFFF;
112
113 for (size_t i = 0; i < length; i++)
114 {
115 crc ^= (uint16_t)data[i];
116 for (int j = 0; j < 8; j++)
117 {
118 if (crc & 0x0001)
119 {
120 crc = (crc >> 1) ^ 0xA001;
121 }
122 else
123 {
124 crc = crc >> 1;
125 }
126 }
127 }
128
129 return crc;
130}

284bool readSingleRegister(uint16_t reg_addr, const char* reg_name, float multiplier, void* target, bool is_float)
285{
286 logDebug("Чтение %s (0x%04X)...", reg_name, reg_addr);
287 uint8_t result = modbus.readHoldingRegisters(reg_addr, 1);
288
289 if (result == modbus.ku8MBSuccess)
290 {
291 uint16_t raw_value = modbus.getResponseBuffer(0);
292
293 if (is_float)
294 {
295 float* float_target = static_cast<float*>(target);
296 *float_target = convertRegisterToFloat(raw_value, multiplier);
297 logDebug("%s: %.2f", reg_name, *float_target);
298 }
299 else
300 {
301 uint16_t* int_target = static_cast<uint16_t*>(target);
302 *int_target = raw_value;
303 logDebug("%s: %d", reg_name, *int_target);
304 }
305 return true;
306 }
307 else
308 {
309 logError("Ошибка чтения %s: %d", reg_name, result);
310 printModbusError(result);
311 return false;
312 }
313}

373static void updateIrrigationFlag(SensorData& d)
374{
375 constexpr uint8_t WIN = 6;
376 static float buf[WIN] = {NAN};
377 static uint8_t idx = 0, filled = 0, persist = 0;
378
379 float baseline = d.humidity;
380 for (uint8_t i = 0; i < filled; ++i) baseline = (buf[i] < baseline) ? buf[i] : baseline;
381
382 bool spike = (filled == WIN) && (d.humidity - baseline >= config.irrigationSpikeThreshold) && (d.humidity > 25.0f);
383 persist = spike ? persist + 1 : 0;
384 if (persist >= 2)
385 {
386 lastIrrigationTs = millis();
387 persist = 0;
388 }
389
390 buf[idx] = d.humidity;
391 idx = (idx + 1) % WIN;
392 if (filled < WIN) ++filled;
393
394 d.recentIrrigation = (millis() - lastIrrigationTs) <= (unsigned long)config.irrigationHoldMinutes * 60000UL;
395}

397static void applyCompensationIfEnabled(SensorData& d)
398{
399 if (!config.flags.calibrationEnabled) return;
400
401 SoilType soil = SoilType::LOAM;
402 SoilProfile profile = SoilProfile::SAND;
403 switch (config.soilProfile)
404 {
405 case 0:
406 soil = SoilType::SAND;
407 profile = SoilProfile::SAND;
408 break;
409 case 1:
410 soil = SoilType::LOAM;
411 profile = SoilProfile::LOAM;
412 break;
413 case 2:
414 soil = SoilType::PEAT;
415 profile = SoilProfile::PEAT;
416 break;
417 case 3:
418 soil = SoilType::CLAY;
419 profile = SoilProfile::CLAY;
420 break;
421 case 4:
422 soil = SoilType::SANDPEAT;
423 profile = SoilProfile::SANDPEAT;
424 break;
425 }
426
427 // Шаг 1: Применяем калибровочную таблицу CSV (лабораторная поверка)
428 float tempCalibrated = CalibrationManager::applyCalibration(d.temperature, profile);
429 float humCalibrated = CalibrationManager::applyCalibration(d.humidity, profile);
430 float ecCalibrated = CalibrationManager::applyCalibration(d.ec, profile);
431 float phCalibrated = CalibrationManager::applyCalibration(d.ph, profile);
432 float nCalibrated = CalibrationManager::applyCalibration(d.nitrogen, profile);
433 float pCalibrated = CalibrationManager::applyCalibration(d.phosphorus, profile);
434 float kCalibrated = CalibrationManager::applyCalibration(d.potassium, profile);
435
436 // Шаг 2: Применяем математическую компенсацию (температурная, влажностная)
437 float ec25 = ecCalibrated / (1.0f + 0.021f * (tempCalibrated - 25.0f));
438 d.ec = correctEC(ec25, tempCalibrated, humCalibrated, soil);
439
440 d.ph = correctPH(phCalibrated, tempCalibrated);
441
442 d.nitrogen = nCalibrated;
443 d.phosphorus = pCalibrated;
444 d.potassium = kCalibrated;
445 correctNPK(tempCalibrated, humCalibrated, d.nitrogen, d.phosphorus, d.potassium, soil);
446
447 // Обновляем остальные значения
448 d.temperature = tempCalibrated;
449 d.humidity = humCalibrated;
450}

456void finalizeSensorData(bool success)
457{
458 sensorData.valid = success;
459 sensorData.last_update = millis();
460
461 if (!success)
462 {
463 logError("❌ Не удалось прочитать один или несколько параметров");
464 return;
465 }
466
467 saveRawSnapshot(sensorData);
468 updateIrrigationFlag(sensorData);
469 applyCompensationIfEnabled(sensorData);
470
471 addToMovingAverage(sensorData, sensorData.temperature, sensorData.humidity, sensorData.ec, sensorData.ph,
472 sensorData.nitrogen, sensorData.phosphorus, sensorData.potassium);
473
474 if (validateSensorData(sensorData))
475 {
476 logSuccess("✅ Все параметры прочитаны и валидны");
477 sensorCache = {sensorData, true, millis()};
478 }
479 else
480 {
481 logWarn("⚠️ Данные прочитаны, но не прошли валидацию");
482 sensorData.valid = false;
483 }
484}

618void addToMovingAverage(SensorData& data, float temp, float hum, float ec, float ph, float n, float p, float k)
619{
620 uint8_t window_size = config.movingAverageWindow;
621 if (window_size < 5) window_size = 5;
622 if (window_size > 15) window_size = 15;
623
624 // ---------- O(1) running sum для каждого параметра ----------
625 static float sum_temp = 0, sum_hum = 0, sum_ec = 0, sum_ph = 0, sum_n = 0, sum_p = 0, sum_k = 0;
626
627 // Если буфер заполнен – вычитаем значение, которое покинет окно
628 if (data.buffer_filled >= window_size)
629 {
630 sum_temp -= data.temp_buffer[data.buffer_index];
631 sum_hum -= data.hum_buffer[data.buffer_index];
632 sum_ec -= data.ec_buffer[data.buffer_index];
633 sum_ph -= data.ph_buffer[data.buffer_index];
634 sum_n -= data.n_buffer[data.buffer_index];
635 sum_p -= data.p_buffer[data.buffer_index];
636 sum_k -= data.k_buffer[data.buffer_index];
637 }
638
639 // Записываем новые значения в буфер и добавляем их к сумме
640 data.temp_buffer[data.buffer_index] = temp;
641 sum_temp += temp;
642 data.hum_buffer[data.buffer_index] = hum;
643 sum_hum += hum;
644 data.ec_buffer[data.buffer_index] = ec;
645 sum_ec += ec;
646 data.ph_buffer[data.buffer_index] = ph;
647 sum_ph += ph;
648 data.n_buffer[data.buffer_index] = n;
649 sum_n += n;
650 data.p_buffer[data.buffer_index] = p;
651 sum_p += p;
652 data.k_buffer[data.buffer_index] = k;
653 sum_k += k;
654
655 // Обновляем индексы кольцевого буфера
656 data.buffer_index = (data.buffer_index + 1) % window_size;
657 if (data.buffer_filled < window_size) data.buffer_filled++;
658
659 uint8_t effective_window = (data.buffer_filled < window_size) ? data.buffer_filled : window_size;
660
661 if (effective_window >= 3 && config.filterAlgorithm == 0)
662 {
663 // Среднее (O(1))
664 data.temperature = sum_temp / effective_window;
665 data.humidity = sum_hum / effective_window;
666 data.ec = sum_ec / effective_window;

698float calculateMovingAverage(float* buffer, uint8_t window_size, uint8_t filled)
699{
700 if (filled == 0) return 0.0;
701
702 // Берем последние filled элементов (или window_size, если filled >= window_size)
703 uint8_t elements_to_use = (filled < window_size) ? filled : window_size;
704
705 // v2.4.1: Используем настраиваемый алгоритм (среднее или медиана)
706 extern Config config;
707
708 if (config.filterAlgorithm == 1)
709 { // FILTER_ALGORITHM_MEDIAN
710 // Создаем временный массив для медианы
711 float temp_values[15]; // Максимальный размер окна
712 for (uint8_t i = 0; i < elements_to_use; i++)
713 {
714 temp_values[i] = buffer[i];
715 }
716
717 // Простая сортировка для медианы
718 for (uint8_t i = 0; i < elements_to_use - 1; i++)
719 {
720 for (uint8_t j = 0; j < elements_to_use - i - 1; j++)
721 {
722 if (temp_values[j] > temp_values[j + 1])
723 {
724 float temp = temp_values[j];
725 temp_values[j] = temp_values[j + 1];
726 temp_values[j + 1] = temp;
727 }
728 }
729 }
730
731 // Возвращаем медиану
732 if (elements_to_use % 2 == 0)
733 {
734 return (temp_values[elements_to_use / 2 - 1] + temp_values[elements_to_use / 2]) / 2.0f;
735 }
736 else
737 {
738 return temp_values[elements_to_use / 2];
739 }
740 }
741 else
742 {
743 // FILTER_ALGORITHM_MEAN (по умолчанию)
744 float sum = 0.0;
745 for (uint8_t i = 0; i < elements_to_use; i++)
746 {
747 sum += buffer[i];
748 }
749 return sum / elements_to_use;
750 }
751}
