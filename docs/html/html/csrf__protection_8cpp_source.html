28String generateCSRFToken()
29{
30 // Используем различные источники энтропии для безопасности
31 String token = "";
32
33 // Компоненты для генерации токена:
34 unsigned long currentTime = millis();
35 uint32_t freeHeap = ESP.getFreeHeap();
36 uint32_t chipId = ESP.getEfuseMac();
37
38 // Добавляем MAC адрес для уникальности
39 String macAddr = WiFi.macAddress();
40 macAddr.replace(":", "");
41
42 // Генерируем токен из различных источников
43 token = String(currentTime, HEX) + String(freeHeap, HEX) + String(chipId, HEX) +
44 macAddr.substring(6); // Последние 6 символов MAC
45
46 // Сохраняем токен и время генерации
47 currentCSRFToken = token;
48 tokenGeneratedTime = currentTime;
49
50 logDebug("CSRF токен сгенерирован: %s (время: %lu)", token.c_str(), currentTime);
51
52 return token;
53}

55bool validateCSRFToken(const String& token)
56{
57 // Если токен пустой или не инициализирован
58 if (token.isEmpty() || currentCSRFToken.isEmpty())
59 {
60 logWarn("CSRF валидация: пустой токен");
61 return false;
62 }
63
64 // Проверяем время жизни токена
65 unsigned long currentTime = millis();
66 if (currentTime - tokenGeneratedTime > CSRF_TOKEN_LIFETIME)
67 {
68 logWarn("CSRF токен истек (время: %lu, генерирован: %lu)", currentTime, tokenGeneratedTime);
69 currentCSRFToken = ""; // Очищаем истекший токен
70 return false;
71 }
72
73 // Проверяем совпадение токенов
74 bool isValid = (token == currentCSRFToken);
75
76 if (isValid)
77 {
78 logDebug("CSRF токен валиден");
79 }
80 else
81 {
82 logWarn("CSRF токен невалиден: ожидался %s, получен %s", currentCSRFToken.c_str(), token.c_str());
83 }
84
85 return isValid;
86}

88String getCSRFHiddenField()

93 generateCSRFToken();

99bool checkCSRFSafety()
100{
101 // GET запросы всегда безопасны (идемпотентные)
102 HTTPMethod method = webServer.method();
103 if (method == HTTP_GET || method == HTTP_HEAD || method == HTTP_OPTIONS)
104 {
105 return true;
106 }
107
108 // POST, PUT, DELETE требуют проверки CSRF
109 String csrfToken = "";
110
111 // Ищем токен в POST параметрах
112 if (webServer.hasArg("csrf_token"))
113 {
114 csrfToken = webServer.arg("csrf_token");
115 }
116
117 // Если токен не найден в POST, ищем в заголовках
118 if (csrfToken.isEmpty() && webServer.hasHeader("X-CSRF-Token"))
119 {
120 csrfToken = webServer.header("X-CSRF-Token");
121 }
122
123 // Логируем попытку доступа
124 String clientIP = webServer.client().remoteIP().toString();
125 String uri = webServer.uri();
126
127 if (csrfToken.isEmpty())
128 {
129 logWarn("CSRF: токен отсутствует для %s %s от %s", methodToString(method).c_str(), uri.c_str(),
130 clientIP.c_str());
131 return false;
132 }
133
134 bool isValid = validateCSRFToken(csrfToken);
135
136 if (!isValid)
137 {
138 logWarn("CSRF: валидация не пройдена для %s %s от %s", methodToString(method).c_str(), uri.c_str(),
139 clientIP.c_str());
140 }
141
142 return isValid;
143}
