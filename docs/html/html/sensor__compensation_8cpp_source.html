38float correctEC(float ecRaw, float T, float theta, SoilType soil)
39{
40 // Шаг 1. Температурная компенсация к 25 °C

56// NPK ---------------------------------------------------------------
57void correctNPK(float T, float theta, float& N, float& P, float& K, SoilType soil)
58{
59 if (theta < 25.0f || theta > 60.0f) return; // валидация – оставляем как есть
60
61 int idx = (int)soil;
62
63 // Температурная коррекция
64 N *= (1.0f - k_t_N[idx] * (T - 25.0f));
65 P *= (1.0f - k_t_P[idx] * (T - 25.0f));
66 K *= (1.0f - k_t_K[idx] * (T - 25.0f));
67
68 // Влажностная коррекция
