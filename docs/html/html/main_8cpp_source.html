59// ✅ Неблокирующая задача мониторинга кнопки сброса
60void resetButtonTask(void* parameter)
61{
62 pinMode(RESET_BUTTON_PIN, INPUT_PULLUP);
63 static unsigned long buttonPressTime = 0;
64 static bool buttonPressed = false;
65
66 while (true)
67 {
68 bool currentState = (digitalRead(RESET_BUTTON_PIN) == LOW);
69
70 if (currentState && !buttonPressed)
71 {
72 // Кнопка только что нажата
73 buttonPressed = true;
74 buttonPressTime = millis();
75 logWarn("Кнопка сброса нажата! Сброс настроек через 2 сек...");
76 }
77 else if (!currentState && buttonPressed)
78 {
79 // Кнопка отпущена раньше времени
80 buttonPressed = false;
81 DEBUG_PRINTLN("Кнопка сброса отпущена");
82 }
83 else if (currentState && buttonPressed && (millis() - buttonPressTime >= BUTTON_HOLD_TIME_MS))
84 {
85 // Кнопка удерживалась 2 секунды
86 logError("Выполняется сброс настроек!");
87 resetConfig();
88 ESP.restart();
89 }
90
91 // ✅ Неблокирующая задержка - проверяем кнопку каждые 50мс
92 vTaskDelay(50 / portTICK_PERIOD_MS);
93 }
94}

98void setup()
99{
100 Serial.begin(115200);
101
102 // *** КРИТИЧЕСКОЕ ОТЛАДОЧНОЕ СООБЩЕНИЕ ***

109 /*-----------------------------------------------------------
110 * Подтверждаем OTA-образ ДО тяжёлой инициализации Wi-Fi/FS.
111 * Нужно успеть до истечения тайм-аутa загрузчика (~5 с).
112 *----------------------------------------------------------*/
113 const esp_partition_t* runningNow = esp_ota_get_running_partition();
114 esp_ota_img_states_t otaStateNow;
115 if (esp_ota_get_state_partition(runningNow, &otaStateNow) == ESP_OK && otaStateNow == ESP_OTA_IMG_PENDING_VERIFY)
116 {
117 logSystem("OTA image pending verify → подтверждаем (ранний этап)");

119 logSuccess("OTA image подтверждена, откат отменён");
120 else
121 logError("Не удалось подтвердить OTA image (ранний этап)!");
122 }
123

181 // Legacy: оставляем старые задачи для поточного обновления sensorData

183 startRealSensorTask();

192 esp_ota_img_states_t ota_state;

211// ✅ Неблокирующий главный цикл с оптимизированными интервалами
212void loop()
213{
214 unsigned long currentTime = millis();
215 esp_task_wdt_reset();
216
217 // Обновление NTP каждые 6 часов
218 if (timeClient && millis() - lastNtpUpdate > 6 * 3600 * 1000)
219 {
220 timeClient->forceUpdate();
221 lastNtpUpdate = millis();
222 logSystem("NTP обновление: %s", timeClient->isTimeSet() ? "OK" : "не удалось");
223 }
224
225 // ✅ Вывод статуса системы каждые 30 секунд (неблокирующий)
226 if (currentTime - lastStatusPrint >= STATUS_PRINT_INTERVAL)
227 {
228 logPrintHeader("СТАТУС СИСТЕМЫ", COLOR_GREEN);
229
230 logUptime();
231 logMemoryUsage();
232 logWiFiStatus();
233 logSystem("Режим датчика: %s", config.flags.useRealSensor ? "РЕАЛЬНЫЙ" : "ЭМУЛЯЦИЯ");
234
235 // Статус данных датчика
236 if (sensorData.valid)
237 {
238 logData("Последние измерения получены %.1f сек назад", (currentTime - sensorData.last_update) / 1000.0);
239 }
240 else
241 {
242 logWarn("Данные датчика недоступны");
243 }
244
245 logPrintSeparator("─", 60);
246 lastStatusPrint = currentTime;
247 }
248
249 // === Проверяем наличие новых данных датчика (НАСТРАИВАЕМО v2.3.0) ===
250 if (sensorData.valid && (currentTime - lastDataPublish >= config.sensorReadInterval)) {
251 pendingMqttPublish = true;
252 pendingThingspeakPublish = true;
253 lastDataPublish = currentTime;
254 DEBUG_PRINTLN("[BATCH] Новые данные помечены для групповой отправки");
255 }
256
257 // ✅ Групповая отправка MQTT (настраиваемо v2.3.0)
258 if (pendingMqttPublish && (currentTime - mqttBatchTimer >= config.mqttPublishInterval)) {
259 publishSensorData();
260 pendingMqttPublish = false;
261 mqttBatchTimer = currentTime;
262 DEBUG_PRINTLN("[BATCH] MQTT данные отправлены группой");
263 }
264
265 // ✅ Групповая отправка ThingSpeak (настраиваемо v2.3.0)
266 if (pendingThingspeakPublish && (currentTime - thingspeakBatchTimer >= config.thingSpeakInterval)) {
267 bool tsOk = sendDataToThingSpeak();
268 if (tsOk) {
269 thingspeakBatchTimer = currentTime; // Сбрасываем таймер только при успешной отправке
270 DEBUG_PRINTLN("[BATCH] ThingSpeak данные отправлены группой");
271 } else {
272 DEBUG_PRINTLN("[BATCH] ThingSpeak отправка не удалась, повтор через следующий интервал");
273 }
274 pendingThingspeakPublish = false;
275 }
276
277 // ✅ Управление MQTT (каждые 100мс)
278 static unsigned long lastMqttCheck = 0;
279 if (currentTime - lastMqttCheck >= 100)
280 {
281 handleMQTT();
282 lastMqttCheck = currentTime;
283 }
284
285 // ✅ Управление WiFi (каждые 20 мс для более высокой отзывчивости веб-интерфейса)
286 static unsigned long lastWiFiCheck = 0;
287 if (currentTime - lastWiFiCheck >= 20)
288 {
289 handleWiFi();
290 lastWiFiCheck = currentTime;
291 }
292
293 // Проверяем OTA раз в час (или при принудительной проверке)
294 static unsigned long lastOtaCheck = 0;
295 if (config.flags.autoOtaEnabled && (currentTime - lastOtaCheck >= 3600000UL))
296 {
297 handleOTA();
298 lastOtaCheck = currentTime;
299 }
300
301 // ✅ Минимальная задержка для стабильности (10мс вместо 100мс)
302 vTaskDelay(10 / portTICK_PERIOD_MS);
303}
